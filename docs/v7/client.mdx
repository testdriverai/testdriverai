## Overview

The `TestDriver` client is the main entry point for the SDK. It manages authentication, sandbox connections, and provides access to all testing methods — element finding, actions, assertions, and provisioning.

## Constructor

```javascript
import TestDriver from 'testdriverai';

const testdriver = new TestDriver(options?)
// or with explicit API key
const testdriver = new TestDriver(apiKey, options?)
```

<ParamField path="apiKey" type="string">
  Your TestDriver API key. If omitted, loaded automatically from the `TD_API_KEY` environment variable or `.env` file.
</ParamField>

<ParamField path="options" type="TestDriverOptions">
  Configuration options for the client.

  <Expandable title="properties">
    <ParamField path="apiKey" type="string">
      API key (alternative to passing as first argument). Falls back to `TD_API_KEY` env var.
    </ParamField>
    
    <ParamField path="apiRoot" type="string" default="https://testdriver-api.onrender.com">
      API endpoint URL. Override for self-hosted deployments. Can also be set via `TD_API_ROOT` env var.
    </ParamField>
    
    <ParamField path="os" type="string" default="linux">
      Operating system for the sandbox: `'windows'` or `'linux'`. Can also be set via `TD_OS` env var.
    </ParamField>
    
    <ParamField path="resolution" type="string" default="1366x768">
      Screen resolution for the sandbox (e.g., `'1920x1080'`, `'1366x768'`).
    </ParamField>
    
    <ParamField path="analytics" type="boolean" default={true}>
      Enable or disable usage analytics.
    </ParamField>
    
    <ParamField path="logging" type="boolean" default={true}>
      Enable or disable console logging.
    </ParamField>
    
    <ParamField path="autoScreenshots" type="boolean" default={false}>
      Automatically capture screenshots before and after each command. Screenshots are saved to `.testdriver/screenshots/<test>/` with descriptive filenames. See [Screenshots](/v7/screenshots) for details.
    </ParamField>
    
    <ParamField path="newSandbox" type="boolean" default={true}>
      Force creation of a new sandbox instead of reusing an existing one.
    </ParamField>
    
    <ParamField path="reconnect" type="boolean" default={false}>
      Reconnect to the last used sandbox instead of creating a new one. When `true`, all [provision](/v7/provision) methods are skipped since the application is already running. Throws error if no previous sandbox exists.
    </ParamField>
    
    <ParamField path="keepAlive" type="number" default={60000}>
      Keep sandbox alive for the specified number of milliseconds after disconnect. Set to `0` to terminate immediately. Useful for debugging or reconnecting to the same sandbox.
    </ParamField>
    
    <ParamField path="preview" type="PreviewMode" default="browser">
      Preview mode for live test visualization:
      - `"browser"` — Opens debugger in default browser (default)
      - `"ide"` — Opens preview in IDE panel (VS Code, Cursor — requires TestDriver extension)
      - `"none"` — Headless mode, no visual preview
    </ParamField>
    
    <ParamField path="headless" type="boolean" default={false}>
      **Deprecated**: Use `preview: "none"` instead.
    </ParamField>
    
    <ParamField path="debugOnFailure" type="boolean" default={false}>
      Keep the sandbox alive when a test fails so you can reconnect and debug interactively. The sandbox stays alive for 5 minutes and connection instructions are logged to the console.
    </ParamField>
    
    <ParamField path="ip" type="string">
      Direct IP address to connect to a running sandbox instance (for self-hosted deployments).
    </ParamField>
    
    <ParamField path="sandboxAmi" type="string">
      Custom AMI ID for the sandbox instance (AWS deployments, e.g., `'ami-1234'`).
    </ParamField>
    
    <ParamField path="sandboxInstance" type="string">
      EC2 instance type for the sandbox (AWS deployments, e.g., `'i3.metal'`).
    </ParamField>
    
    <ParamField path="dashcam" type="boolean" default={true}>
      Enable or disable [Dashcam](/v7/dashcam) video recording.
    </ParamField>
    
    <ParamField path="cache" type="boolean | CacheConfig" default={true}>
      Enable or disable element [caching](/v7/cache), or provide advanced threshold configuration.
      
      <Expandable title="advanced config">
        <ParamField path="enabled" type="boolean" default={true}>
          Enable or disable caching.
        </ParamField>
        
        <ParamField path="cacheKey" type="string">
          Cache key for grouping cached results. Auto-generated from the calling test file's content hash when not provided.
        </ParamField>
        
        <ParamField path="thresholds" type="object">
          Fine-tune cache matching.
          
          <Expandable title="properties">
            <ParamField path="find" type="object">
              Thresholds for `find()` operations.
              
              <Expandable title="properties">
                <ParamField path="screen" type="number" default={0.05}>
                  Pixel diff threshold for screen comparison (0–1). `0.05` = 5% diff allowed.
                </ParamField>
                
                <ParamField path="element" type="number" default={0.8}>
                  OpenCV template match threshold for element matching (0–1). `0.8` = 80% correlation required.
                </ParamField>
              </Expandable>
            </ParamField>
            
            <ParamField path="assert" type="number" default={0.05}>
              Pixel diff threshold for `assert()` operations (0–1). `0.05` = 5% diff allowed.
            </ParamField>
          </Expandable>
        </ParamField>
      </Expandable>
    </ParamField>
    
    <ParamField path="cacheKey" type="string">
      **Legacy**: Use `cache.cacheKey` instead. Cache key for element finding operations.
    </ParamField>
    
    <ParamField path="redraw" type="boolean | RedrawConfig" default={false}>
      Enable or disable screen-change (redraw) detection. Disabled by default since v7.3 — individual commands still use their own [redraw](/v7/redraw) options.
      
      <Expandable title="advanced config">
        <ParamField path="enabled" type="boolean" default={false}>
          Enable or disable global redraw detection.
        </ParamField>
        
        <ParamField path="thresholds" type="object">
          Threshold configuration.
          
          <Expandable title="properties">
            <ParamField path="screen" type="number | false" default={0.05}>
              Pixel diff threshold (0–1). Set to `false` to disable screen redraw detection.
            </ParamField>
            
            <ParamField path="network" type="boolean" default={false}>
              Enable or disable network activity monitoring.
            </ParamField>
          </Expandable>
        </ParamField>
      </Expandable>
    </ParamField>
    
    <ParamField path="environment" type="object">
      Additional environment variables to pass to the sandbox as key-value pairs.
    </ParamField>
    
    <ParamField path="ai" type="AIConfig">
      Global AI sampling configuration. Controls how the AI model generates responses for `find()` verification and `assert()` calls. Can be overridden per-command.
      
      <Expandable title="properties">
        <ParamField path="model" type="string">
          AI model to use (e.g., `"gpt-4o"`, `"claude-3"`).
        </ParamField>
        
        <ParamField path="provider" type="string">
          AI provider name.
        </ParamField>
        
        <ParamField path="apiKey" type="string">
          Provider API key for BYOK (bring your own key).
        </ParamField>
        
        <ParamField path="baseURL" type="string">
          Custom endpoint URL for the AI provider.
        </ParamField>
        
        <ParamField path="thinking" type="boolean">
          Enable thinking/reasoning mode for the AI model.
        </ParamField>
        
        <ParamField path="temperature" type="number">
          Controls randomness in AI responses. `0` = deterministic (best for verification), higher values = more creative.
        </ParamField>
        
        <ParamField path="top" type="object">
          Nucleus and top-k sampling parameters.
          
          <Expandable title="properties">
            <ParamField path="p" type="number">
              Top-P (nucleus sampling). Limits token choices to smallest set whose cumulative probability exceeds P. Range: 0–1.
            </ParamField>
            
            <ParamField path="k" type="number">
              Top-K sampling. Limits token choices to top K most likely tokens. `1` = always pick most likely. `0` = disabled.
            </ParamField>
          </Expandable>
        </ParamField>
      </Expandable>
    </ParamField>
  </Expandable>
</ParamField>

### Examples

```javascript
import TestDriver from 'testdriverai';

// Minimal — API key from .env
const testdriver = new TestDriver();

// Specify OS and resolution
const testdriver = new TestDriver({
  os: 'windows',
  resolution: '1920x1080',
});

// With explicit API key
const testdriver = new TestDriver('your-api-key', {
  os: 'windows',
});

// With AI config and caching
const testdriver = new TestDriver({
  ai: { temperature: 0, top: { p: 0.9, k: 40 } },
  cache: {
    thresholds: {
      find: { screen: 0.03, element: 0.85 },
      assert: 0.03,
    },
  },
});

// Self-hosted
const testdriver = new TestDriver({
  ip: '192.168.1.100',
  apiRoot: 'https://my-testdriver.internal.com',
});
```

## Lifecycle Methods

### auth()

Authenticate with the TestDriver API. Exchanges your API key for a session token.

```javascript
await testdriver.auth()
```

**Returns:** `Promise<void>`

<Note>
  You must call `auth()` before `connect()`. When using the Vitest hook or presets, this is handled automatically.
</Note>

### connect()

Connect to a sandbox environment. Creates or reconnects to a virtual machine where tests run.

```javascript
await testdriver.connect(options?)
```

<ParamField path="options" type="ConnectOptions">
  Connection options that override constructor settings.

  <Expandable title="properties">
    <ParamField path="sandboxId" type="string">
      Existing sandbox ID to reconnect to.
    </ParamField>
    
    <ParamField path="newSandbox" type="boolean" default={false}>
      Force creation of a new sandbox.
    </ParamField>
    
    <ParamField path="reconnect" type="boolean" default={false}>
      Reconnect to last used sandbox.
    </ParamField>
    
    <ParamField path="ip" type="string">
      Direct IP address (self-hosted).
    </ParamField>
    
    <ParamField path="os" type="string">
      Override OS for this connection.
    </ParamField>
    
    <ParamField path="sandboxAmi" type="string">
      AMI for AWS sandbox.
    </ParamField>
    
    <ParamField path="sandboxInstance" type="string">
      EC2 instance type.
    </ParamField>
    
    <ParamField path="preview" type="PreviewMode" default="browser">
      Preview mode: `"browser"`, `"ide"`, or `"none"`.
    </ParamField>
    
    <ParamField path="reuseConnection" type="boolean" default={true}>
      Reuse existing connection if connected.
    </ParamField>
    
    <ParamField path="keepAlive" type="number" default={60000}>
      Keep sandbox alive after disconnect (ms).
    </ParamField>
  </Expandable>
</ParamField>

**Returns:** `Promise<Object>` — Sandbox instance details including `instanceId`, `ip`, `vncPort`.

```javascript
// Basic connection
await testdriver.connect();

// Reconnect to existing sandbox
await testdriver.connect({ sandboxId: 'existing-sandbox-id-123' });

// Self-hosted
await testdriver.connect({ ip: '192.168.1.100' });
```

### ready()

Convenience method that calls `auth()` + `connect()` in sequence.

```javascript
await testdriver.ready()
```

**Returns:** `Promise<void>`

```javascript
const testdriver = new TestDriver({ os: 'linux' });
await testdriver.ready(); // Authenticates and connects
```

### disconnect()

Disconnect from the sandbox and clean up resources.

```javascript
await testdriver.disconnect()
```

**Returns:** `Promise<void>`

```javascript
afterAll(async () => {
  await testdriver.disconnect();
});
```

<Warning>
  Always call `disconnect()` in cleanup hooks (`afterAll`, `finally` blocks). This prevents orphaned sandboxes from consuming resources.
</Warning>

## Instance Information

### getInstance()

Get the current sandbox instance details.

```javascript
const instance = testdriver.getInstance()
```

**Returns:** `Object | null` — Sandbox instance info (`instanceId`, `ip`, etc.)

### getSessionId()

Get the current session ID for tracking and debugging.

```javascript
const sessionId = testdriver.getSessionId()
```

**Returns:** `string | null`

## Logging & Events

### setLogging()

Toggle console logging at runtime.

```javascript
testdriver.setLogging(enabled)
```

<ParamField path="enabled" type="boolean" required>
  Whether to enable logging.
</ParamField>

```javascript
testdriver.setLogging(false);
await testdriver.disconnect();
testdriver.setLogging(true);
```

### getEmitter()

Get the [EventEmitter2](https://github.com/EventEmitter2/EventEmitter2) instance for custom event handling.

```javascript
const emitter = testdriver.getEmitter()
```

**Returns:** `EventEmitter2` — Supports wildcard events with `:` delimiter.

```javascript
const emitter = testdriver.getEmitter();

emitter.on('command:start', (data) => {
  console.log('Command started:', data);
});

emitter.on('command:success', (data) => {
  console.log('Command succeeded:', data);
});

emitter.on('command:error', (error) => {
  console.error('Command failed:', error);
});
```

See [Events](/v7/events) for the full list of event namespaces.

### getLogs() / clearLogs()

Access and manage the internal log buffer. Logs are stored in JSONL format for S3 upload.

```javascript
const logData = testdriver.getLogs()    // string (JSONL)
testdriver.clearLogs()
```

## Test Run Tracking

### createTestRun()

Create a new test run in the TestDriver API.

```javascript
const run = await testdriver.createTestRun(options)
```

**Returns:** `Promise<object>` — Test run metadata.

### completeTestRun()

Mark a test run as complete.

```javascript
await testdriver.completeTestRun(options)
```

### recordTestCase()

Record an individual test case result.

```javascript
await testdriver.recordTestCase(options)
```

### setTestContext()

Set the Vitest test context for automatic test metadata tracking.

```javascript
testdriver.setTestContext(context)
```

<ParamField path="context" type="object" required>
  Vitest task context (`context.task`).
</ParamField>

## Unawaited Promise Detection

The SDK tracks the last returned promise. If a new command is called before the previous promise resolves, it logs a warning:

```
⚠️ Previous TestDriver command was not awaited!
```

Always `await` every TestDriver command to prevent race conditions:

```javascript
// ❌ Missing await — will trigger warning
testdriver.find('button').click();
testdriver.type('hello');

// ✅ Correct
await testdriver.find('button').click();
await testdriver.type('hello');
```

## Complete Example

```javascript
import { beforeAll, afterAll, describe, it } from 'vitest';
import TestDriver from 'testdriverai';

describe('My Test Suite', () => {
  let testdriver;

  beforeAll(async () => {
    testdriver = new TestDriver({
      os: 'windows',
      resolution: '1366x768',
      logging: true,
      cache: {
        thresholds: {
          find: { screen: 0.05, element: 0.8 },
          assert: 0.05,
        },
      },
    });
    
    const emitter = testdriver.getEmitter();
    emitter.on('log:info', (msg) => console.log('[INFO]', msg));
    
    await testdriver.auth();
    const instance = await testdriver.connect();
    console.log('Connected:', instance.instanceId);
  });

  afterAll(async () => {
    await testdriver.disconnect();
  });

  it('runs a test', async () => {
    await testdriver.provision.chrome({ url: 'https://example.com' });
    await testdriver.find('Get Started button').click();
  });
});
```

## Best Practices

<AccordionGroup>
  <Accordion title="Reuse sandboxes across tests">
    Use `beforeAll`/`afterAll` to create one sandbox per test suite. This significantly reduces execution time vs creating a sandbox per test.
  </Accordion>
  
  <Accordion title="Handle connection errors gracefully">
    Wrap `connect()` in try-catch to handle network issues or quota limits:
    
    ```javascript
    try {
      await testdriver.connect();
    } catch (error) {
      console.error('Connection failed:', error.message);
      throw error;
    }
    ```
  </Accordion>
  
  <Accordion title="Always disconnect in cleanup">
    Use `afterAll` or try-finally blocks to ensure `disconnect()` is called even if tests fail:
    
    ```javascript
    afterAll(async () => {
      await testdriver.disconnect();
    });
    ```
  </Accordion>
  
  <Accordion title="Use environment variables for API keys">
    Never hardcode API keys:
    
    ```bash .env
    TD_API_KEY=your_api_key_here
    ```
    
    ```javascript
    // API key loaded automatically
    const testdriver = new TestDriver();
    ```
  </Accordion>
  
  <Accordion title="Use debugOnFailure for debugging">
    When a test fails and you need to inspect the sandbox interactively:
    
    ```javascript
    const testdriver = new TestDriver({
      debugOnFailure: true
    });
    ```
    
    The sandbox stays alive for 5 minutes after failure, and connection instructions are logged.
  </Accordion>
</AccordionGroup>
